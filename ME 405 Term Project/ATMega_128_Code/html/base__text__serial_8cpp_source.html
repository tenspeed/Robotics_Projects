173 /** This method writes a character to the serial port as a text string showing the
174  * 8-bit unsigned number in that character. If the format control term 'ascii' was
175  * called prior to this printing, the character will be printed as an ASCII character.
176  * @param num The 8-bit number or character to be sent out
177  */
178 
179 base_text_serial& base_text_serial::operator<< (unsigned char num)
180 {
181  unsigned char temp_char; // Temporary storage for a nibble
182 
183  if (print_ascii)
184  {
185  putchar (num);
186  }
187  else if (base == 2)
188  {
189  for (unsigned char bmask = 0x80; bmask != 0; bmask >>= 1)
190  {
191  if (num & bmask) putchar ('1');
192  else putchar ('0');
193  }
194  }
195  else if (base == 16)
196  {
197  temp_char = (num >> 4) & 0x0F;
198  putchar ((temp_char > 9) ? temp_char + ('A' - 10) : temp_char + '0');
199  temp_char = num & 0x0F;
200  putchar ((temp_char > 9) ? temp_char + ('A' - 10) : temp_char + '0');
201  }
202  else
203  {
204  char out_str[9];
205  utoa ((unsigned int)num, out_str, base);
206  puts (out_str);
207  }
208 
209  return (*this);
210 }

368 /** This method writes a single-precision floating point number to the serial port in
369  * exponential format (always with the 'e' notation). It calls the utility function
370  * __ftoa_engine, which is hiding in the AVR libraries, used by the Xprintf()
371  * functions when they need to convert a float into text.
372  * @param num The floating point number to be sent out
373  */
374 
375 base_text_serial& base_text_serial::operator<< (float num)
376 {
377  char digits = precision;
378  char buf[20];
379  char* p_buf = buf;
380 
381  int exponent = __ftoa_engine ((double)num, buf, digits, 16);
382  uint8_t vtype = *p_buf++;
383  if (vtype & FTOA_NAN)
384  {
385  *this << " NaN";
386  return (*this);
387  }
388 
389  // Display the sign if it's negative
390  if (vtype & FTOA_MINUS)
391  putchar ('-');
392 
393  // Show the mantissa
394  putchar (*p_buf++);
395  if (digits)
396  putchar ('.');
397  while ((digits-- > 0) && *p_buf)
398  putchar (*p_buf++);
399 
400  // Now display the exponent
401  putchar ('e');
402  if (exponent > 0)
403  putchar ('+');
404  *this << exponent;
405 }

473  * numbers to binary, octal, decimal, or hexadecimal. Also, and endline is provided

479 base_text_serial& base_text_serial::operator<< (ser_manipulator new_manip)
480 {
481  switch (new_manip)
482  {
483  case (bin): // Print integers in binary
484  base = 2;
485  break;
486  case (oct): // Print integers in octal
487  base = 8;
488  break;
489  case (dec): // Print integers in decimal
490  base = 10;
491  break;
492  case (hex): // Print integers in hexadecimal
493  base = 16;
494  break;
495  case (ascii): // Print chars as ASCII letters, etc.
496  print_ascii = true;
497  break;
498  case (numeric): // Print chars as numbers
499  print_ascii = false;
500  break;
501  case (endl): // Send an end-of-line
502  puts (ENDL_STYLE);
503  break;
504  case (clrscr): // Send a clear-screen code
505  clear_screen ();
506  break;
507  case (send_now): // Send whatever's in the send buffer
508  transmit_now ();
509  break;
510  case (_p_str): // The next string is in program memory
511  pgm_string = true;
512  };
513 
514  return (*this);
515 }

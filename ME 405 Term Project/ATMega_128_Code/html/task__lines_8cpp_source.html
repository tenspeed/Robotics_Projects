67 /** run is the main method in task_lines. It takes care of the math involved in cutting a line into segments. It
68 * first takes the initial and final x&y coordinates, uses the pythagorean theorem to find the length of the
69 * actual line, determins how many segments to break the line up into based on it's overall length, calculates
70 * the slope of the line, and then uses the slope along with the change in the x&y coordinates to determine
71 * if the line passes through the origin (type 1), is vertical (type 2), is horizontal (type 4), or general
72 * and none of the above (type 3). The reason for categorizing what type of line you're drawing is because
73 * the algorithm is slightly different for each line type and certain lines can be drawn much faster by using
74 * a different algorithm, especially lines which pass through the origin. task_lines then breaks the line into
75 * smaller segments, keeps track of what segment the polar plotter is currently on, increments setpoints as
76 * each one is reached, and uses the point object to physically move the pen to the next setpoint. All of this
77 * is accomplished using a switch-case structure and flags which are raised and lowered depending on what needs
78 * to happen.
79 * @param state This method uses STL_task to transition its state
80 */
81 char task_lines::run(char state)
82 {
83  switch (state)
84  {
85  /// State 0 is a wait state. Is a new line requested?
86  case 0:
87  // start a new line
88  if (divide_lines == true)
89  {
90  current_segment = 0;
91  divide_lines = false;
92  if (signature == true)
93  {
94  send_home = true;
95  return(9);
96  }
97  else
98  {
99  return(1);
100  }
101  }
102  // do nothing
103  else
104  {
105  return (STL_NO_TRANSITION);
106  }
107  break;
108 
109  /// State 1 does some initial x-y to r-theta calculations
110  case 1:
111  // calculate total change in x and y
112  SUMdx = ((int32_t)(xf - x0));
113  SUMdy = ((int32_t)(yf - y0));
114 
115  // prepare for pythagorean theorum
116  SUMx_squared = pow((float)SUMdx, 2L);
117  SUMy_squared = pow((float)SUMdy, 2L);
118 
119  //length of line in tenths of inches
120  Seg_Total = (uint16_t)sqrt(SUMx_squared + SUMy_squared);
121 
122  // calculate segment lengths
123  dx = ((1000*SUMdx)/Seg_Total)/400;
124  dy = ((1000*SUMdy)/Seg_Total)/400;
125 
126  // calculate slope
127  if (SUMdx == 0)
128  {
129  //line is vertical, slope undefined (infinity)
130  line_type = 2;
131  }
132  else
133  {
134  slope = (float)SUMdy/SUMdx;
135 
136  // does this line go through the origin?
137  float intercept = y0 - (slope*x0);
138 
139  if ((intercept < 1L) && (intercept > -1L))
140  {
141  // this line passes through the origin and has a positive slope
142  line_type = 1;
143  }
144  else if (SUMdy == 0)
145  {
146  // horizontal line
147  line_type = 4;
148  }
149  else
150  {
151  // a general line that does not pass through the origin and is not vertical or horizontal
152  line_type = 3;
153  }
154  }
155  return(2);
156  break;
157 
158  /// State 2 starts movement towards initial point
159  case 2:
160  Initial_Point->Get_There(x0,y0);
161  return(3);
162  break;
163 
164  /// State 3 waits for attainment of initial point
165  case 3:
166  counter++;
167  if ( Initial_Point->Are_We_There() )
168  {
169  // if not yet started making line, drop pen, raise flag, and reset counter
170  if (moving == false)
171  {
172  counter = 0;
173  moving = true;
174  operate_pen->Fine_Line();
175  }
176  }
177 
178  //waiting counter allows time for pen to get down, then starts motor initialization
179  if ((counter >= 50UL) && moving)
180  {
181  moving = false;
182  // set line type
183  switch (line_type)
184  {
185  // line through origin
186  case 1:
187  return (4);
188  break;
189 
190  // vertical line
191  case 2:
192  SUMdy = y0 + dy;
193  current_segment = 0;
194  return(5);
195  break;
196 
197  // normal line
198  case 3:
199  SUMdx = ((int32_t)x0) + ((int32_t)dx);
200  SUMdy = ((int32_t)y0) + ((int32_t)dy);
201  current_segment = 0;
202  return(6);
203  break;
204 
205  // horizontal line
206  case 4:
207  SUMdx = x0 + dx;
208  current_segment = 0;
209  return(7);
210  break;
211  }
212  }
213  return (STL_NO_TRANSITION);
214  break;
215 
216  /// state 4 draws line which would pass through origin
217  case 4:
218  if (moving == false)

246

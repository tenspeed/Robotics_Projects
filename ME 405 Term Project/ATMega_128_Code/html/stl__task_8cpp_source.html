105 /** This method is called by the main task loop to try to run the task. If the task is
106  * in the waiting state, it checks to see if it's time to run yet; if it's in the
107  * suspended state, it doesn't. The task shouldn't be in the running state, because
108  * this method is used by the cooperative scheduler, not the pre-emptive one.
109  * @param til_next_time A pointer to a time stamp which will hold next time this task
110  * next needs to run, or NULL (default) if this data isn't needed
111  * @return The operational state of the task after it has been given this chance to
112  * run
113  */
114 
115 task_op_state stl_task::schedule (time_stamp* til_next_time)
116 {
117  char next_state; // State to which a task will transition
118  time_stamp the_time; // Somewhere to store a time measurement
119 
120  switch (op_state)
121  {
122  // If the task has been suspended, don't bother trying to run it, and don't
123  // bother sending back the next run time because this task is suspended
124  case (TASK_SUSPENDED):
125  return (TASK_SUSPENDED);
126 
127  // If the task needs to run, check if it needs to run now; if so, run it
128  case (TASK_WAITING):
129  // Find out what the current real time is from the system timer
130  the_time = the_timer.get_time_now ();
131 
132  // If it's not time to run the task yet, exit without running it
133  if (next_run_time > the_timer.get_time_now ())
134  {
135  if (til_next_time != NULL)
136  {
137  *til_next_time = next_run_time;
138  }
139  return (TASK_WAITING);
140  }
141  // If we get here, it is time to run the task; just continue into the
142  // task_pending section below, which will cause the task to run right now
143 
144  case (TASK_PENDING):
145  // Set the state to waiting for the next time interval. If the task needs
146  // to run again immediately, run_again_ASAP() will be called within the
147  // run() method, causing the state to be set to TASK_PENDING instead
148  op_state = TASK_WAITING;
149  #ifdef STL_PROFILE // If execution time profiling is
150  start_profiler (); // activated, start timing
151  #endif
152  next_state = run (current_state); // Call the run() method
153  #ifdef STL_PROFILE
154  end_profiler (); // End execution time measurement
155  #endif
156  if (next_state != STL_NO_TRANSITION) // Detect state transition if any
157  { // has occurred
158  #ifdef STL_TRACE
159  GLOB_DEBUG ("T" << serial_number << ":" << current_state << "-"
160  << next_state << endl);
161  #endif
162  current_state = next_state; // Go to next state next time
163  }
164 
165  // If the op-state is TASK_WAITING now, then run_again_ASAP() was not
166  // called, and the task will wait until its time to run again has arrived
167  if (op_state == TASK_WAITING)
168  {
169  next_run_time += interval;
170  if (til_next_time != NULL)
171  {
172  *til_next_time = next_run_time;
173  }
174  }
175 
176  // If the op-state isn't TASK_WAITING, someone called run_again_ASAP(), so
177  // the op-state is TASK_PENDING and we need to run again right away
178  return (op_state);
179 
180  // If the operational state is anything else, there has been a serious error
181  default:
182  error_stop ("Illegal operational state");
183  break;
184  };
185 
186 // else if (op_state == blocked) // Blocked tasks are not currently
187 // return (false); // implemented
188 
189 }

299 /** This method ends a measurement run, computing the duration of the run and saving
300  * that duration in the statistical counters as necessary.
301  */
302 
303 void stl_task::end_profiler (void)
304 {
305  // Compute the duration measured for this run
306  time_stamp duration = the_timer.get_time_now () - start_time;
307 
308  // A duration less than zero is impossible, so it must be wrong
309  if (duration < 0)
310  {
311  return;
312  }
313 
314  // If we have a maximum record, save it
315  if (duration > max_time)
316  max_time = duration;
317 
318  // If this is the first run, it's the minimum; otherwise, check it
319  if (runs == 0L)
320  min_time = duration;
321  else
322  if (duration < min_time)
323  min_time = duration;
324 
325  // Add to the counters which are used to find the average
326  run_time_sum += duration;
327  runs++;
328 }

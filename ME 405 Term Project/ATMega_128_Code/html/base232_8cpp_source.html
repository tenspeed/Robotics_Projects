56 base232::base232 (unsigned int baud_rate, unsigned char port_number)
57 {
58  // If we're compiling for a chip with UCSR0A defined, it has dual serial ports
59  // (examples are ATmega324P and ATmega128). Set up Port 0 or Port 1
60  #if defined UCSR0A
61  if (port_number == 0)
62  {
63  p_UDR = &UDR0;
64  p_USR = &UCSR0A;
65  p_UCR = &UCSR0B;
66  UCSR0B = (1 << RXEN0) | (1 << TXEN0);
67  UCSR0C = (1 << UCSZ01) | (1 << UCSZ00); // | (1 << USBS0);
68  UBRR0H = 0x00;
69  UBRR0L = calc_baud_div (baud_rate);
70  #ifdef UART_DOUBLE_SPEED // Activate double speed mode
71  UCSR0A |= U2X0; // if required
72  #endif
73  mask_UDRE = (1 << UDRE0);
74  mask_RXC = (1 << RXC0);
75  mask_TXC = (1 << TXC0);
76  }
77  else // The port number isn't 0, so it presumably must be serial port 1
78  {
79  #if defined UCSR1A
80  p_UDR = &UDR1;
81  p_USR = &UCSR1A;
82  p_UCR = &UCSR1B;
83  UCSR1B = (1 << RXEN1) | (1 << TXEN1);
84  UCSR1C = (1 << UCSZ11) | (1 << UCSZ10); // | (1 << USBS1);
85  UBRR1H = 0x00;
86  UBRR1L = calc_baud_div (baud_rate);
87  #ifdef UART_DOUBLE_SPEED // If double-speed macro has been defined,
88  UCSR1A |= U2X1; // turn on double-speed operation
89  #endif
90  mask_UDRE = (1 << UDRE1);
91  mask_RXC = (1 << RXC1);
92  mask_TXC = (1 << TXC1);
93  #endif // UCSR1A
94  }
95  // We're compiling for a chip which has no USCR0A, so it should have only one
96  // serial port. Ignore the port number and set up the only serial port available
97  #else // no UCSR0A
98  // The more sophisticated chips, such as ATmega8 and ATmega32, have UCSRA
99  // and UCSRB and UCSRC registers which are set up appropriately
100  #ifdef UCSRA
101  p_UDR = &UDR;
102  p_USR = &UCSRA;
103  p_UCR = &UCSRB;
104  UCSRB = (1 << RXEN) | (1 << TXEN);
105  UCSRC = (1 << URSEL) | (1 << UCSZ1) | (1 << UCSZ0); // | (1 << USBS0);
106  UBRRH = 0x00;
107  UBRRL = calc_baud_div (baud_rate);
108  #ifdef UART_DOUBLE_SPEED // Activate double speed mode
109  UCSRA |= U2X; // if required
110  #endif
111  mask_UDRE = (1 << UDRE);
112  mask_RXC = (1 << RXC);
113  mask_TXC = (1 << TXC);
114  // Older, simpler chips such as AT90S2313 have only the UCR register
115  #else // no UCSRA
116  p_UDR = &UDR;
117  p_USR = &USR;
118  p_UCR = &UCR;
119  UCR = (1 << RXEN) | (1 << TXEN); // 0x18 for mode N81
120  UBRR = calc_baud_div (baud_rate);
121  mask_UDRE = (1 << UDRE);
122  mask_RXC = (1 << RXC);
123  mask_TXC = (1 << TXC);
124  #endif // UCSRA
125  #endif // UCSR0A
126 
127  // Read the data register to ensure that it's empty
128  port_number = *p_UDR;
129  port_number = *p_UDR;
130 }

2 /** \file queue.h
3  * This file implements a simple queue (or circular buffer). Data which is placed in
4  * the queue is saved for future reading. The queue class is implemented as a
5  * template; this means that a queue can be configured to store data of almost any
6  * type.
7  *
8  * Usage:
9  * The template specifies the type of data being stored as qType, the type of
10  * integer used for the index as qIndexType, and the number of data items in the
11  * queue as qSize. First choose the type of data being stored; this may be simple
12  * data types such as characters or integers, but it may also be something more
13  * complex such as objects of a class (if there's enough memory to hold them all).
14  * Then look at the number of items in the queue. If there are less than 255, use
15  * a character as qIndexType; otherwise use a regular 16-bit integer. For example,
16  * a queue holding 100 long integers (which would use 400 bytes for the data plus
17  * a few more bytes for the indices) could be declared as
18  * "queue<long, char, 100> my_queue".
19  *

121 template <class qType, class qIndexType, qIndexType qSize>
122 bool queue<qType, qIndexType, qSize>::put (qType data)
123 {
124  // Check if the buffer is already full
125  if (how_full >= qSize)
126  return true;
127 
128  // OK, there's room in the buffer so add the data in
129  buffer[i_put] = data;
130  if (++i_put >= qSize)
131  i_put = 0;
132  how_full++;
133 
134  return false;
135 }

145 template <class qType, class qIndexType, qIndexType qSize>
146 bool queue<qType, qIndexType, qSize>::jam (qType data)
147 {
148  // Write the data and move the write pointer to the next element
149  buffer[i_put] = data;
150  if (i_put >= qSize)
151  i_put = 0;
152 
153  // Check if the buffer is already full; if so, the read index has to be moved so
154  // that it points to the oldest unread data, which isn't the data written now
155  if (how_full >= qSize)
156  {
157  if (++i_get >= qSize)
158  i_get = 0;
159  return true;
160  }
161  else
162  {
163  how_full++; // If we get here, the buffer isn't full
164  return false;
165  }
166 }

187  how_full--; // There's now one less item in the buffer
188  }

208 /** This overloaded array subscript operator implements subscripts in the queue. The
209  * index is defined relative to the location at which the oldest data in the queue
210  * was written. This operator returns the (index)-th item in the queue, or (-1)
211  * typecast to the queue data type if there aren't enough items in the queue to get
212  * to the given index. The retreived data is not removed from the queue.
213  * @param index The array index into the queue at which to get a data item
214  * @return The data which is at the given index
215  */
216 
217 template <class qType, class qIndexType, qIndexType qSize>
218 qType queue<qType, qIndexType, qSize>::operator[] (qIndexType index)
219 {
220  // Check if there's data written at the given location
221  if (index > how_full)
222  return ((qType)(-1));
223 
224  // Find an index pointing to the correct location in the queue
225  qIndexType getIndex = i_get + index;
226  if (getIndex > qSize)
227  getIndex -= qSize;
228 
229  // Get the data at that index location
230  return (buffer[getIndex]);
231 }

276 /** This function sends a command packet to the SD card.
277  * @param cmd The command byte to be sent
278  * @param arg The argument to be sent along with the command
279  */
280 
281 extern "C" // This function is compiled as C, not C++
282 {
283  static BYTE MM_send_cmd (BYTE cmd, DWORD arg)
284  {
285  BYTE n, res;
286 
287  if (cmd & 0x80) /* ACMD<n> is the command sequence of CMD55-CMD<n> */
288  {
289  cmd &= 0x7F;
290  res = MM_send_cmd (CMD55, 0);
291  if (res > 1) return res;
292  }
293 
294  /* Select the card and wait for ready */
295  DESELECT ();
296  SELECT ();
297  n = MM_wait_ready ();
298  if (n != 0xFF)
299  {
300 // GLOB_DEBUG (PMS ("MM_send_cmd: timeout, got ") << n << endl);
301  return 0xFF;
302  }
303 
304  /* Send command packet */
305  MM_xmit_spi (cmd); /* Start + Command index */
306  MM_xmit_spi ((BYTE)(arg >> 24)); /* Argument[31..24] */
307  MM_xmit_spi ((BYTE)(arg >> 16)); /* Argument[23..16] */
308  MM_xmit_spi ((BYTE)(arg >> 8)); /* Argument[15..8] */
309  MM_xmit_spi ((BYTE)arg); /* Argument[7..0] */
310  n = 0x01; /* Dummy CRC + Stop */
311  if (cmd == CMD0) n = 0x95; /* Valid CRC for CMD0(0) */
312  if (cmd == CMD8) n = 0x87; /* Valid CRC for CMD8(0x1AA) */
313  MM_xmit_spi(n);
314 
315  /* Receive command response */
316  if (cmd == CMD12) MM_rcvr_spi (); /* Skip a stuff byte when stop reading */
317  n = 10; /* Wait for a valid response in timeout of 10 attempts */
318  do
319  res = MM_rcvr_spi();
320  while ((res & 0x80) && --n);
321 
322  return res; /* Return with the response value */
323  }
324 } // Extern "C"

328 /** This function initializes the SD card.
329  * @param drive_number The drive number; this function works with one SD card only,
330  * so the drive number is ignored
331  */
332 
333 extern "C" // This function is compiled as C, not C++
334 {
335  DSTATUS disk_initialize (BYTE drive_number)
336  {
337  BYTE n, ty, cmd, ocr[4];
338 
339  MM_power_on (); // Force socket power ON
340  FCLK_SLOW (); // Slow the clock down
341 
342  DESELECT (); // Make sure card is not selected
343  for (n = 10; n; n--) // 80 dummy clocks
344  MM_rcvr_spi ();
345 
346  ty = 0;
347  n = MM_send_cmd (CMD0, 0);
348  if (n == 1) // Enter Idle state
349  {
350  int32_t retry_cnt = 10000L;
351  if (MM_send_cmd (CMD8, 0x1AA) == 1) // SDC ver 2.00
352  {
353  for (n = 0; n < 4; n++)
354  ocr[n] = MM_rcvr_spi ();
355  if (ocr[2] == 0x01 && ocr[3] == 0xAA) // Card can work at vdd in 2.7-3.6V
356  {
357  // ACMD41 with HCS bit
358  while ((--retry_cnt > 0) && MM_send_cmd (ACMD41, 1UL << 30));
359  if ((--retry_cnt > 0) && MM_send_cmd (CMD58, 0) == 0)
360  { /* Check CCS bit */
361  for (n = 0; n < 4; n++)
362  ocr[n] = MM_rcvr_spi ();
363  ty = (ocr[0] & 0x40) ? CT_SD2 | CT_BLOCK : CT_SD2; /* SDv2 */
364  }
365  }
366  }
367  else
368  { /* SDv1 or MMCv3 */
369  if (MM_send_cmd (ACMD41, 0) <= 1)
370  {
371  ty = CT_SD1; cmd = ACMD41; /* SDv1 */
372  }
373  else
374  {
375  ty = CT_MMC; cmd = CMD1; /* MMCv3 */
376  }
377  // Wait for leaving idle state
378  while ((--retry_cnt > 0) && MM_send_cmd (cmd, 0));
379 
380  // Select R/W block length
381  if ((retry_cnt <= 0) || MM_send_cmd (CMD16, 512) != 0)
382  ty = 0;
383  }
384  }
385 // else
386 // {
387 // GLOB_DEBUG (PMS ("disk_initialize(): got ") << hex << n << dec
388 // << PMS (", not 1") << endl);
389 // }
390 
391  CardType = ty;
392  MM_release_spi ();
393 
394  if (ty) /* Initialization succeded */
395  {
396  Stat &= ~STA_NOINIT; /* Clear STA_NOINIT */
397  FCLK_FAST ();
398  }
399  else
400  { /* Initialization failed */
401  MM_power_off ();
402  }
403 
404  return (Stat);
405  }
406 } // extern "C"

435 extern "C" // This function is compiled as C, not C++
436 {
437  DRESULT disk_read (BYTE drive_number, BYTE *buff, DWORD sector, BYTE count)
438  {
439  if (!count) return RES_PARERR;
440  if (Stat & STA_NOINIT) return RES_NOTRDY;
441 
442  if (!(CardType & CT_BLOCK)) sector *= 512; /* Convert LBA to byte address if needed */
443

456  }

516 /** This function is used to carry out various miscellaneous I/O functions with the
517  * SD card. The functions which are supported are the following:
518  * \li CTRL_SYNC - Ensure that any pending write process has finished
519  * \li GET_SECTOR_COUNT - Get the number of sectors on the disk
520  * \li GET_SECTOR_SIZE - Get the sector size
521  * \li GET_BLOCK_SIZE - Get the block size (should always be 512 for SD cards)
522  * \li MMC_GET_CSD -
523  * \li MMC_GET_CID -
524  * \li MMC_GET_OCR -
525  * \li MMC_GET_SDSTAT -
526  *
527  * @param drive_number The drive number; this function works with one SD card only,
528  * so the drive number is ignored
529  * @param ctrl The control code which tells which function is called for
530  * @param buff A pointer to a buffer used for sector I/O to and from the disk
531  */
532 
533 extern "C" // This function is compiled as C, not C++
534 {
535  DRESULT disk_ioctl (BYTE drive_number, BYTE ctrl, void *buff)
536  {
537  DRESULT res;
538  BYTE n, csd[16], *ptr = (uint8_t*)buff;
539  WORD csize;
540 
541  if (Stat & STA_NOINIT) return RES_NOTRDY;
542 
543  res = RES_ERROR;
544  switch (ctrl)
545  {
546  /* Make sure that pending write process has been finished */
547  case CTRL_SYNC :
548  SELECT ();
549  if (MM_wait_ready() == 0xFF)
550  res = RES_OK;
551  break;
552 
553  /* Get number of sectors on the disk (DWORD) */
554  case GET_SECTOR_COUNT :
555  if ((MM_send_cmd (CMD9, 0) == 0) && MM_rcvr_datablock (csd, 16))
556  {
557  if ((csd[0] >> 6) == 1) /* SDC ver 2.00 */
558  {
559  csize = csd[9] + ((WORD)csd[8] << 8) + 1;
560  *(DWORD*)buff = (DWORD)csize << 10;
561  }
562  else /* MMC or SDC ver 1.XX */
563  {
564  n = (csd[5] & 15) + ((csd[10] & 128) >> 7)
565  + ((csd[9] & 3) << 1) + 2;
566  csize = (csd[8] >> 6) + ((WORD)csd[7] << 2)
567  + ((WORD)(csd[6] & 3) << 10) + 1;
568  *(DWORD*)buff = (DWORD)csize << (n - 9);
569  }
570  res = RES_OK;
571  }
572  break;
573 
574  /* Get sectors on the disk (WORD) */
575  case GET_SECTOR_SIZE :
576  *(WORD*)buff = 512;
577  res = RES_OK;
578  break;
579 
580  /* Get erase block size in unit of sectors (DWORD) */
581  case GET_BLOCK_SIZE :
582  if (CardType & CT_SD2) /* SDC ver 2.00 */
583  {
584  if (MM_send_cmd(ACMD13, 0) == 0)
585  { /* Read SD status */
586  MM_rcvr_spi ();
587  if (MM_rcvr_datablock (csd, 16)) /* Read partial block */
588  {
589  /* Purge trailing data */
590  for (n = 64 - 16; n; n--) MM_rcvr_spi ();
591  *(DWORD*)buff = 16UL << (csd[10] >> 4);
592  res = RES_OK;
593  }
594  }
595  }
596  else /* SDC ver 1.XX or MMC */
597  {
598  if ((MM_send_cmd (CMD9, 0) == 0) && MM_rcvr_datablock(csd, 16))
599  {
600  if (CardType & CT_SD1) /* SDC ver 1.XX */
601  {
602  *(DWORD*)buff = (((csd[10] & 63) << 1)
603  + ((WORD)(csd[11] & 128) >> 7)
604  + 1) << ((csd[13] >> 6) - 1);
605  }
606  else
607  { /* MMC */
608  *(DWORD*)buff = ((WORD)((csd[10] & 124) >> 2) + 1)
609  * (((csd[11] & 3) << 3) + ((csd[11] & 224) >> 5) + 1);
610  }
611  res = RES_OK;
612  }
613  }
614  break;
615 
616  /* Receive CSD as a data block (16 bytes) */
617  case MMC_GET_CSD :
618  if (MM_send_cmd (CMD9, 0) == 0 && MM_rcvr_datablock (ptr, 16))
619  res = RES_OK;
620  break;
621 
622  /* Receive CID as a data block (16 bytes) */
623  case MMC_GET_CID :
624  if (MM_send_cmd (CMD10, 0) == 0
625  && MM_rcvr_datablock(ptr, 16))

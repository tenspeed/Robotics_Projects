2 /** \file stl_timer.cpp
3  * This file contains a class which runs a task timer whose resolution is one
4  * microsecond. The timer is used to help schedule the execution of tasks' run()
5  * functions. The functions can be run from a main loop by checking periodically
6  * if a task time has expired, or they can be called directly from the timer
7  * interrupt service routine in this file, or they can be called from some other
8  * hardware interrupt service routine; in the last case this file isn't involved.
9  *
10  * Revisions:
11  * \li 08-07-2007 JRR Created this file as daytimer.* with 1 second interrupts
12  * \li 08-08-2007 JRR Added event triggers
13  * \li 12-23-2007 JRR Made more general by allowing faster interrupt rates
14  * \li 01-05-2008 JRR Converted from time-of-day version to microsecond version
15  * \li 03-27-2008 JRR Added operators + and - for time stamps
16  * \li 03-31-2008 JRR Merged in stl_us_timer (int, long) and set_time (int, long)
17  * \li 05-15-2008 JRR Changed to use Timer 3 so Timer 1 can run motor PWM's
18  * \li 05-31-2008 JRR Changed time calculations to use CPU_FREQ_MHz from Makefile
19  * \li 01-04-2009 JRR Now uses CPU_FREQ_Hz (rather than MHz) for better precision
20  * \li 11-24-2009 JRR Changed CPU_FREQ_Hz to F_CPU to match AVR-LibC's name

322 /** This constructor creates a daytime task timer object. It sets up the hardware timer
323  * to count at ~1 MHz and interrupt on overflow. Note that this method does not enable
324  * interrupts globally, so the user must call sei() at some point to enable the timer
325  * interrupts to run and time to actually be measured.
326  */
327 
328 task_timer::task_timer (void)
329 {
330  #ifdef TCNT3 // If Timer 3 exists, we'll use it
331  TCCR3A = 0x00; // Set to normal 16-bit counting mode
332  TCCR3B = (1 << CS31); // Set prescaler to main clock / 8
333  #ifdef ETIMSK // For ATmega128
334  ETIMSK |= (1 << TOIE3); // Set Timer 3 overflow interrupt enable
335  #endif
336  #ifdef TIMSK3 // For ATmega1281
337  TIMSK3 |= (1 << TOIE3); // Set Timer 3 overflow interrupt enable
338  #endif
339  #else // No Timer 3, so use Timer 1
340  TCCR1A = 0x00; // Set to normal 16-bit counting mode
341  TCCR1B = (1 << CS11); // Set prescaler to main clock / 8
342  #ifdef TIMSK1 // If there's a TIMSK1, set it
343  TIMSK1 |= (1 << TOIE1); // Enable Timer 1 overflow interrupt
344  #else // If no TIMSK1, there's a TIMSK
345  TIMSK |= (1 << TOIE1); // Enable Timer 1 overflow interrupt
346  #endif
347  #endif
348 }

359 void task_timer::save_time_stamp (time_stamp& the_stamp)
360 {
361  uint8_t temp_sreg = SREG; // Store interrupt flag status
362  the_stamp.data.half[0] = TMR_TCNT_REG; // Get hardware count
363  cli (); // Prevent interruption
364  the_stamp.data.half[1] = ust_overflows; // Get overflow counter data
365  SREG = temp_sreg; // Re-enable interrupts if they were on
366 }

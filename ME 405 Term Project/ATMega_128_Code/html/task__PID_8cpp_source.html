68 /** run is a two state PID controller. The first state is simply a non moving state which checks to see if
69 * movement has been requested. State two incorporates proportional, integral and differential feedback
70 * into a control loop which updates the duty cycle of a motor (one motor per PID object).
71 */
72 char task_PID::run(char state)
73 {
74  switch (state)
75  {
76  /** State 0 is a non moving state. It doesn't move.
77  */
78  case 0:
79  // if homing underway, leave and do nothing
80  if (homing)
81  {
82  return (STL_NO_TRANSITION);
83  }
84 
85  func_UPDATE_MOTOR->update_duty_cycle(motor_num, 0);
86  if (giddyup == true)
87  {
88  return (1); //transition to state 1 to do PID
89  }
90  else
91  return (STL_NO_TRANSITION); // continue to be off
92  break;
93  /** State 1 is the PID task. It gets a current encoder reading (ensuring no transfer error) and then
94  * calculates new duty and sets the duty cycle.
95  */
96  case 1:
97  bool Checksum_Error_flag = 1; // mock an encoder checksum fault
98  while (Checksum_Error_flag) // get encoder readings until no checksum fault
99  {
100  func_READ_ENCODER -> Initiate(motor_num);
101  Checksum_Error_flag = func_READ_ENCODER -> Get_Checksum_flag();
102  }
103  encoder = func_READ_ENCODER -> Get_Encoder();
104 
105  // calculate error and saturate
106  int32_t error_now = (Set_Point - encoder);
107 
108  int32_t proportional_error = (int32_t)(((int64_t)error_now * (int64_t)K_p)/K_P_DIVISOR);
109 
110  int32_t E_sum_now = error_now + E_sum_old;
111 
112  // ensure integral portion of feedback loop is within reason
113  if (E_sum_now > INTEGRAL_SATURATE)
114  E_sum_now = INTEGRAL_SATURATE;
115  else if (E_sum_now < (-INTEGRAL_SATURATE))
116  E_sum_now = (-INTEGRAL_SATURATE);
117 
118  int32_t integral_error = (int32_t)(((int64_t)E_sum_now * (int64_t)K_i)/K_I_DIVISOR);
119 
120 
121  //int32_t differential_error = ((error_now - prev_error) * (K_d))/K_D_DIVISOR;
122 
123  //archive data
124  E_sum_old = E_sum_now;
125  prev_error = error_now;
126 
127  int32_t OUTPUT = (proportional_error + integral_error);// + differential_error);
128 
129  // If error_now yields a negative value, set direction as reverse
130  if (OUTPUT < 0)
131  {
132  func_UPDATE_MOTOR->set_mode(motor_num, 2); // make motor go CCW
133  }
134  // If error_now is positive, make motor go forward
135  else
136  func_UPDATE_MOTOR->set_mode(motor_num, 1);
137 
138  // Ensure OUTPUT is positive before updating the duty cycle
139  if (OUTPUT < 0)
140  {
141  OUTPUT *= (-1);
142  }
143  else
144  {}
145 
146  // Saturate duty cycle
147  if (OUTPUT > DUTY_CYCLE_SATURATE)
148  {
149  OUTPUT = DUTY_CYCLE_SATURATE;
150  }
151  duty_cycle = (uint8_t) OUTPUT;
152 
153  // update speed of motor
154  func_UPDATE_MOTOR->update_duty_cycle(motor_num, duty_cycle);
155 
156  // If somebody pressed the stop button, transition to the stopped state
157  if (giddyup == false)
158  {
159  return (0);
160  }
161  // transition to new set point if appropriate
162  if ( (error_now <= 400) && (error_now >= -400) )
163  {
164  dummy++;
165  if (dummy >= 10L)
166  {
167  are_we_there_yet = true;
168  dummy = 0;
169  }
170  }
171  return (STL_NO_TRANSITION);
172  }

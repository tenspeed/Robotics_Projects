101  /** Initiate communicates with another SPI device. Each encoder value is 32 bits or 4 bytes. The data
102  * transfer across the SPI bus happens one byte at a time. Each byte is stored in the union encoder_data.
103  * After the complete encoder value has been recieved, one more data transfer occurs to get the checksum
104  * from the slave device. The checksum for the received data is calculated and then compared to the checksum
105  * sent by the slave. If the checksums don't match, there has been a transfer error and the pointer to the
106  * encoder variable in PID_test is not updated. If the checksums match, then the pointer to the encoder
107  * variable gets updated with the new encoder position.
108  * @param encoder_select Initiate must be fed which channel (motor) you desire encoder ticks from
109  */
110  void Master::Initiate (uint8_t encoder_select)
111  {
112  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
113  encoder_num = encoder_select;
114  PORTA &= ~(1<<PIN7); // Activate SS
115  if (encoder_num == 1) // send 01 command if encoder_num = 1
116  {
117  Check_Sum_Master = 0x01; // put command into check sum 1
118  SPDR = 0x01; // send transfer request command 1
119  while(!(SPCR && (1<<SPIF))) {} // wait for transfer
120  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
121  }
122  else if (encoder_num == 2) // send 02 command if encoder_num = 2
123  {
124  Check_Sum_Master = 0x02; // put command into check sum 2
125  SPDR = 0x02; // send transfer request command 2
126  while(!(SPCR && (1<<SPIF))) {} // wait for transfer
127  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
128  }
129 
130  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
131  SPDR = 0xFF; // send junk, get first byte
132  while(!(SPCR && (1<<SPIF))) {} // wait for transfer
133  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
134  encoder_data.parts[0] = SPDR;
135  Check_Sum_Master += encoder_data.parts[0];
136 
137  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
138  SPDR = 0xFF; // send junk, get second byte
139  while(!(SPCR && (1<<SPIF))) {} // wait for transfer
140  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
141  encoder_data.parts[1] = SPDR;
142  Check_Sum_Master += encoder_data.parts[1];
143 
144  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
145  SPDR = 0xFF; // send junk, get third byte
146  while(!(SPCR && (1<<SPIF))) {} // wait for transfer
147  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
148  encoder_data.parts[2] = SPDR;
149  Check_Sum_Master += encoder_data.parts[2];
150 
151  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
152  SPDR = 0xFF; // send junk, get fourth byte
153  while(!(SPCR && (1<<SPIF))) {} // wait for transfer
154  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
155  encoder_data.parts[3] = SPDR;
156  Check_Sum_Master += encoder_data.parts[3];
157 
158  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
159  SPDR = 0xFF; // send junk, get check sum
160  while(!(SPCR && (1<<SPIF))) {} // wait for transfer
161  while (++dummy < WAIT) {} dummy=0; // hold it up a sec
162  Slave_Sum = SPDR;
163  if (Check_Sum_Master == Slave_Sum)
164  {
165  error_flag = false; // no error occurred
166  encoder = encoder_data.whole;
167  }
168  else
169  {
170  error_flag = true; // transmission error occurred!
171  Checksum_Error++;
172  }
173  PORTA |= (1<<PIN7); // Drive SS high, disableing
174  Check_Sum_Master = 0; // Reset Check_Sum_Master before exiting
175  Slave_Sum = 0; // Reset Slave_Sum before exiting
176  encoder_data.whole = 0; // Reset the union before exiting
177  }

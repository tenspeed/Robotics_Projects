67 /** run is the main method in Go_Home and it performs the actual homing operation of the 2 degree
68 * of freedom system.
69 * @param state: STL_task is used for scheduling
70 */
71 char Go_Home::run(char state)
72 {
73  switch (state)
74  {
75  /// State 0 does nothing unless homing is requested by user key press or task_lines
76  case 0:
77  if ((Home_Request) || (LINES->wanna_go_home()))
78  {
79  // if home requested by lines, enable interrupts
80  Home_Request = true;
81 
82  //if cart not already homed, engage cart motor towards home, disable cart's PID
83  if (PORTE && PIN4)
84  {
85  // set motor direction towards r=0 and engage motor.
86  Vtec_wins->set_mode (1, 2);
87  Wipe_Master1->Request_Home(true);
88  Vtec_wins->update_duty_cycle (1, 150);
89 
90  }
91  return(1);
92  }
93  else
94  return(STL_NO_TRANSITION);
95  break;
96 
97  /// State 1 initiates movement of the arm towards home if cart has reached r=0
98  case 1:
99  // when cart is home, shut off cart motor, engage arm motor towards home, and disable arm's PID
100  if (cart_switch)
101  {
102  // Shut off cart motor
103  Vtec_wins->update_duty_cycle (1, 0);
104 
105  // if arm not already homed, send it home.
106  if (PORTE && PIN5)
107  {
108  // set motor direction towards theta=0 and engage motor.
109  Vtec_wins->set_mode (2, 2);
110  Wipe_Master2->Request_Home(true);
111  Vtec_wins->update_duty_cycle (2, 255);
112  }
113  return(2);
114  }
115  return(STL_NO_TRANSITION);
116  break;
117 
118  /// State 2 concludes homing once arm reaches THETA=0.
119  case 2:
120  if (arm_switch)
121  {
122  // shut off cart motor
123  Vtec_wins->update_duty_cycle (2, 0);
124 
125  //clear all homing bools, and encoder and error counts
126  Wipe_Slave->CLEAR(1);
127  Wipe_Slave->CLEAR(2);
128  Wipe_Master1->CLEAR();
129  Wipe_Master2->CLEAR();
130  arm_switch = false;
131  cart_switch = false;
132  Home_Request = false;
133 
134  // Tell PIDs that they have control of the motors again
135  Wipe_Master1->Request_Home(false);
136  Wipe_Master2->Request_Home(false);
137  LINES->ok_were_home();
138  return(0);
139  }
140  return(STL_NO_TRANSITION);
141  break;
142  }
143 }

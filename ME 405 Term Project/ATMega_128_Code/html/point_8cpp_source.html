53 /** The run method moves the pen to a specific location. It has two modes. It can just move the pen somewhere
54 * or it can move the pen somewhere and then make a dot.
55 */
56 void point::run()
57 {
58  switch (point_state)
59  {
60  /// State 0 is a wait state. Is a point or move requested?
61  case 0:
62  if (Make_Point || Go_To)
63  {
64  x_squared = pow((float)X, 2L);
65  y_squared = pow((float)Y, 2L);
66  encoder_goal_r = (int32_t)sqrt(x_squared + y_squared);
67  encoder_goal_r = encoder_goal_r*2350; //took this step out of the sqrt() function to minimize chance for math errors
68 
69  // calculate theta position, including oddballs THETA = 0 or 90 degrees
70  if ((X == 0) && Y)
71  {
72  encoder_goal_theta = 925000;
73  }
74  else if (Y == 0)
75  {
76  encoder_goal_theta = 0;
77  }
78  else
79  {
80  encoder_goal_theta = (int32_t)( atan(((float)Y)/((float)X)) * 588873);
81  }
82 
83  // set setpoints and engage motors
84  PID_1->set_setpoint(encoder_goal_r);
85  PID_2->set_setpoint(encoder_goal_theta);
86  PID_1->go();
87  PID_2->go();
88  // state transition
89  point_state = 1;
90  }
91  break;
92 
93  /// State 1 checks if correct location is achieved yet, if so, make dot if necessary
94  case 1:
95  // if both motors are done moving ...
96  if ( (PID_1->At_Seg_End()) && (PID_2->At_Seg_End()) )
97  {
98  // if in point mode lower pen, start counting from 0
99  if (Make_Point)
100  {
101  counting = 0;
102  Pen_Servo->Fine_Line();
103  point_state = 2;
104  }
105 
106  // must be in go-to mode, no further action necessary
107  else
108  {
109  Go_To = false;
110  point_state = 0;
111  PID_1->stop();
112  PID_2->stop();
113  }
114  }
115 
116  break;
117 
118  /// State 2 allows time for pen to move to the down position
119  case 2:
120  if (counting >= 5000UL)
121  {
122  Pen_Servo->Pen_Up();
123  Make_Point = false;
124  counting = 0;
125  point_state = 0;
126  }
127  counting++;
128  break;
129  }
130 }

54 /** This macro computes a value for the baud rate divisor from the desired baud rate
55  * and the CPU clock frequency. The CPU clock frequency should have been set in the
56  * macro F_CPU, which is normally configured in the Makefile. The divisor is
57  * calculated as (frequency, MHz / (16 * baudrate)), unless the USART is running in
58  * double-speed mode, in which case the baud rate divisor is twice as big.
59  */
60 
61 #ifdef UART_DOUBLE_SPEED
62  #define calc_baud_div(baud_rate) ((F_CPU) / (16UL * (baud_rate)))
63 #else
64  #define calc_baud_div(baud_rate) ((F_CPU) / (8UL * (baud_rate)))
65 #endif

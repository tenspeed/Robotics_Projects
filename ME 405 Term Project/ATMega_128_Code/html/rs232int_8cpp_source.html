63 /** This method sets up the AVR UART for communications. It calls the base_text_serial
64  * constructor, which prepares to convert numbers to text strings, and the base232
65  * constructor, which does the work of setting up the serial port. Note that the user
66  * program must call sei() somewhere to enable global interrupts so that this driver
67  * will work. It is not called in this constructor because it's common to construct

70  * @param baud_rate The desired baud rate for serial communications. Default is 9600

75 rs232::rs232 (unsigned int baud_rate, unsigned char port_number)
76  : base_text_serial (), base232 (baud_rate, port_number)
77 {
78  // Save the number of the serial port, 0 or 1
79  port_num = port_number;
80 
81  // If we're compiling for a chip with UCSR0A defined, it has dual serial ports
82  // (examples are ATmega324P and ATmega128). Set up Port 0 or Port 1
83  #if defined UCSR0A // Serial port number 0
84  if (port_number == 0)
85  {
86  UCSR0B |= (1 << RXCIE0); // Receive complete interrupt enable
87 
88  // Allocate some memory for the receiver buffer and reset the indices
89  rcv0_buffer = new uint8_t[RSINT_BUF_SIZE];
90  rcv0_read_index = 0;
91  rcv0_write_index = 0;
92  }
93  else // Serial port number 1
94  {
95  #if defined UCSR1A
96  UCSR1B |= (1 << RXCIE1); // Receive complete interrupt enable
97 
98  // Allocate some memory for the receiver buffer and reset the indices
99  rcv1_buffer = new uint8_t[RSINT_BUF_SIZE];
100  rcv1_read_index = 0;
101  rcv1_write_index = 0;
102  #endif // UCSR1A
103  }
104  // We're compiling for a chip which doesn't define UCSR0A; assume it has only one
105  // serial port.
106  #else
107  UCSRB |= (1 << RXCIE); // Receive complete interrupt enable
108 
109  // Allocate some memory for the receiver buffer and reset the indices
110  rcv0_buffer = new uint8_t[RSINT_BUF_SIZE];
111  rcv0_read_index = 0;
112  rcv0_write_index = 0;
113  #endif
114 
115  // The Xiphos 1.0 board may need the pullup activated on the RXD1 line in order to
116  // use the XBee radio module
117  #ifdef XIPHOS_HACKS
118  if (port_number == 1)
119  PORTD |= 0x04;
120  #endif
121 }

125 /** This method sends one character to the serial port. It waits until the port is
126  * ready, so it can hold up the system for a while. It times out if it waits too
127  * long to send the character; you can check the return value to see if the character
128  * was successfully sent, or just cross your fingers and ignore the return value.
129  * Note 1: It's possible that at slower baud rates and/or higher processor speeds,
130  * this routine might time out even when the port is working fine. A solution would
131  * be to change the count variable to an integer and use a larger starting number.
132  * Note 2: Fixed! The count is now an integer and it works at lower baud rates.
133  * @param chout The character to be sent out
134  * @return True if everything was OK and false if there was a timeout
135  */
136 
137 bool rs232::putchar (char chout)
138 {
139  // Now wait for the serial port transmitter buffer to be empty
140  for (unsigned int count = 0; ((*p_USR & mask_UDRE) == 0); count++)
141  {
142  if (count > UART_TX_TOUT)
143  return (false);
144  }
145 
146  // Clear the TXCn bit so it can be used to check if the serial port is busy. This
147  // check needs to be done prior to putting the processor into sleep mode. Oddly,
148  // the TXCn bit is cleared by writing a one to its bit location
149  *p_USR |= mask_TXC;
150 
151  // The CTS line is 0 and the transmitter buffer is empty, so send the character
152  *p_UDR = chout;
153  return (true);
154 }

170 /** This method gets one character from the serial port, if one is there. If not, it
171  * waits until there is a character available. This can sometimes take a long time
172  * (even forever), so use this function carefully. One should almost always use
173  * check_for_char() to ensure that there's data available first.
174  * @return The character which was found in the serial port receive buffer
175  */
176 
177 char rs232::getchar (void)
178 {
179  uint8_t recv_char; // Character read from the queue
180 
181  // Wait until there's a character in the receiver queue
182  #ifdef UCSR0A // If this is a dual-port chip
183  if (port_num == 0)
184  {
185  while (rcv0_read_index == rcv0_write_index);
186  recv_char = rcv0_buffer[rcv0_read_index];
187  if (++rcv0_read_index >= RSINT_BUF_SIZE)
188  rcv0_read_index = 0;
189  }
190  else // This is port 1 of a dual-port chip
191  {
192  #if defined UCSR1A
193  while (rcv1_read_index == rcv1_write_index);
194  recv_char = rcv1_buffer[rcv1_read_index];
195  if (++rcv1_read_index >= RSINT_BUF_SIZE)
196  rcv1_read_index = 0;
197  #endif // UCSR1A
198  }
199  #else // This chip has only one serial port
200  while (rcv0_read_index == rcv0_write_index);
201  recv_char = rcv0_buffer[rcv0_read_index];
202  if (++rcv0_read_index >= RSINT_BUF_SIZE)
203  rcv0_read_index = 0;
204  #endif
205 
206  return (recv_char);
207 }

219 {
220  #ifdef UCSR1A // If this is a dual-port chip
221  if (port_num == 0)
222  return (rcv0_read_index != rcv0_write_index);
223  else
224  return (rcv1_read_index != rcv1_write_index);

248 ISR (RSI_CHAR_RECV_INT_0)
249 {
250  // When this ISR is triggered, there's a character waiting in the USART data reg-
251  // ister, and the write index indexes the place where that character should go
252 
253  #if defined UCSR0A // If this is a dual-serial-port chip (ATmega324P, 128, etc.)
254  rcv0_buffer[rcv0_write_index] = UDR0;
255  #else // If this chip has only a single serial port (ATmega8, 32, etc.)
256  rcv0_buffer[rcv0_write_index] = UDR;
257  #endif
258 
259  // Increment the write pointer
260  if (++rcv0_write_index >= RSINT_BUF_SIZE)
261  rcv0_write_index = 0;
262 
263  // If the write pointer is now equal to the read pointer, that means we've just
264  // overwritten the oldest data. Increment the read pointer so that it doesn't seem
265  // as if the buffer is empty
266  if (rcv0_write_index == rcv0_read_index)
267  if (++rcv0_read_index >= RSINT_BUF_SIZE)
268  rcv0_read_index = 0;
269 }
